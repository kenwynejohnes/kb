# Showing current status of files in the repo
git status
# Change log
git log
# Add a new file to the index
git add <new_file>
# Commit changes
git commit -m (for short comments)
# Sync changes to the server
git push
# Show branch information
git branch
# See what has changed since the last commit
git diff
# If you created a branch locally, in order to push the changes to the server, you have to:
git push --set-upstream origin fix-misprint
